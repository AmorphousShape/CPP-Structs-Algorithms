296 if (j == 0) {
297 table[j][k] = 0;

299 table[j][k] = table[j - 1][k];

322 void choose() {
323
324 // Start at the bottom right corner of the table and work backwards
325 int k = max;
326 for (int i = n; i > 0; i--) {
327 if (table[i-1][k] == table[i][k]) {
328 objects[i] = false;
329 } else {
330 objects[i] = true;
331 k -= weights[i];
332 }

457 string build(int j, int k) {
458
459 // Base Cases
460 if (j == 0 || k == 0) {
461 return "";
462 }
463
464 // If the characters are the same, add it to the LCS
465 if (s1[j-1] == s2[k-1]) {
466 return build(j - 1, k - 1) + s1[j-1];
467 }
468
469 // If the characters are different, backtrack
470 if (L[j][k-1] == L[j][k]) {
471 return build(j, k - 1);
472 }
473 return build(j - 1, k);
474 }

503 string get() {
504
505 if (built) {
506 return foundLCS;
507 }
508
509 // Populate vector with 0s
510 for (int i = 0; i <= s1.length(); i++) {
511 vector<int> temp;
512 for (int j = 0; j <= s2.length(); j++) {
513 temp.push_back(0);
514 }
515 L.push_back(temp);
516 }
517
518 // LCS Algorithm
519 for (int j = 1; j <= s1.length(); j++) {
520 for (int k = 1; k <= s2.length(); k++) {
521 if (s1[j-1] == s2[k-1]) {
522 L[j][k] = L[j-1][k-1] + 1;
523 } else {
524 L[j][k] = max(L[j-1][k], L[j][k-1]);
525 }
526 }
527 }
528
529 // Set foundLCS for future use and update built
530 foundLCS = build(s1.length(), s2.length());
531 built = true;
532
533 return foundLCS;
534 }

695 void calcCosts() {

697 for (int i = 0; i < n - L; i++) {

699 cost->data[i][j] = INT_MAX;

703 cost->data[i][j] = q;

808void worldSeries(int n, double *aProb) {
809
810 // Create matrix
811 double x[n+1][n+1];
812
813 for (int i = 0; i <= n; i++) {
814 for (int j = 0; j <= n; j++) {
815
816 // Base cases
817 if (i == 0 && j == 0) {
818 x[0][0] = 1;
819 } else if (i == n && j == n) {
820 x[n][n] = 0;
821
822 // Fill in matrix
823 } else if ((i != 0 && j == 0) || (i == n && j != n)) {
824 x[i][j] = x[i - 1][j] * aProb[i + j];
825 } else if ((i == 0 && j != 0) || (i != n && j == n)) {
826 x[i][j] = x[i][j - 1] * (1 - aProb[i + j]);
827 } else {
828 x[i][j] = x[i - 1][j] * aProb[i + j] + x[i][j - 1] * (1 - aProb[i + j]);
829 }
830 }
831 }
832
833 // Print matrix
834 for (int i = 0 ; i < (n+1)*(n+1); i++) {
835 cout << x[i/(n+1)][i%(n+1)] << " ";
836 if (i % (n+1) == n) {
837 cout << endl;
838 }
839 }
