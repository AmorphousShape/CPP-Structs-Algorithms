386 void merge(int l, int m, int r) {
387 int sub1 = m - l + 1, sub2 = r - m;
388
389 elmtype *linfo = new elmtype[sub1], *rinfo = new elmtype[sub2];
390
391 for (int i = 0; i < sub1; i++) {
392 int tempIndex = (l + i + start) % cap;
393 linfo[i] = info[tempIndex];
394 }
395 for (int j = 0; j < sub2; j++) {
396 int tempIndex = (m + j + 1 + start) % cap;
397 rinfo[j] = info[tempIndex];
398 }
399
400 int indexL = 0, indexR = 0, indexM = l;
401
402 // Primary Merge Loop
403 while (indexL < sub1 && indexR < sub2) {
404 if (linfo[indexL] <= rinfo[indexR]) {
405 info[(indexM + start) % cap] = linfo[indexL];
406 indexL++;
407 } else {
408 info[(indexM + start) % cap] = rinfo[indexR];
409 indexR++;
410 }
411 indexM++;
412 }
413
414 // Merge Remaining Lefts
415 while (indexL < sub1) {
416 info[(indexM + start) % cap] = linfo[indexL];
417 indexL++;
418 indexM++;
419 }
420
421 // Merge Remaining Rights
422 while (indexR < sub2) {
423 info[(indexM + start) % cap] = rinfo[indexR];
424 indexR++;
425 indexM++;
426 }
427
428 // Clean Up and Return
429 delete[] linfo;
430 delete[] rinfo;
431 return;
432 };

489 elmtype KthSmallest(elmtype *arr, int l, int r, int k) {
490 elmtype pivot = arr[(l + r) / 2];
491 int pos = partition(arr, l, r, pivot);
492
493 if (pos - l == k - 1) {
494 return arr[pos];
495 } else if (pos - l > k - 1) {
496 return KthSmallest(arr, l, pos - 1, k);
497 } else {
498 return KthSmallest(arr, pos + 1, r, k - pos + l - 1);
499 }
500 }

514 elmtype KthSmallestWC(elmtype *arr, int l, int r, int k) {
515 int n = r - l + 1, i;
516 elmtype median[(n + 4) / 5];
517 for (i = 0; i < n / 5; i++) {
518 median[i] = getMedian(arr + l + i * 5, 5);
519 }
520 if (n > i * 5) {
521 median[i] = getMedian(arr + l + i * 5, n % 5);
522 i++;
523 }
524 elmtype medOfMed =
525 (i == 1) ? median[i - 1] : KthSmallestWC(median, 0, i - 1, i / 2);
526 int pos = partition(arr, l, r, medOfMed);
527
528 if (pos - l == k - 1) {
529 return arr[pos];
530 } else if (pos - l > k - 1) {
531 return KthSmallestWC(arr, l, pos - 1, k);
532 } else {
533 return KthSmallestWC(arr, pos + 1, r, k - pos + l - 1);
534 }
535 }

542 elmtype getMedian(elmtype *arr, int n) {
543 sort(arr, n);

559 int partition(elmtype arr[], int l, int r, elmtype pivot) {
560 int i;
561 for (i = l; i < r; i++) {
562 if (arr[i] == pivot) {
563 break;
564 }
565 }
566 swap(&arr[i], &arr[r]);
567 i = l;
568 for (int j = l; j <= r - 1; j++) {
569 if (arr[j] <= pivot) {
570 swap(&arr[i], &arr[j]);
571 i++;
572 }
573 }
574 swap(&arr[i], &arr[r]);
575 return i;
576 }

583 void sort(elmtype *arr, int n) {
584 for (int i = 0; i < n; i++) {
585 elmtype *smallest = &arr[0];
586 for (int j = i + 1; j < n; j++) {
587 if (arr[j] < (*smallest)) {
588 smallest = &arr[j];
589 }
590 }
591 if (&arr[i] != smallest) {
592 swap(&arr[i], smallest);
593 }
594 }
