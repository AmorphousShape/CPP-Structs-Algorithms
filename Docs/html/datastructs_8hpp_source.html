384 void merge(int l, int m, int r) {
385 int sub1 = m - l + 1, sub2 = r - m;
386
387 elmtype *linfo = new elmtype[sub1], *rinfo = new elmtype[sub2];
388
389 for (int i = 0; i < sub1; i++) {
390 int tempIndex = (l + i + start) % cap;
391 linfo[i] = info[tempIndex];
392 }
393 for (int j = 0; j < sub2; j++) {
394 int tempIndex = (m + j + 1 + start) % cap;
395 rinfo[j] = info[tempIndex];
396 }
397
398 int indexL = 0, indexR = 0, indexM = l;
399
400 // Primary Merge Loop
401 while (indexL < sub1 && indexR < sub2) {
402 if (linfo[indexL] <= rinfo[indexR]) {
403 info[(indexM + start) % cap] = linfo[indexL];
404 indexL++;
405 } else {
406 info[(indexM + start) % cap] = rinfo[indexR];
407 indexR++;
408 }
409 indexM++;
410 }
411
412 // Merge Remaining Lefts
413 while (indexL < sub1) {
414 info[(indexM + start) % cap] = linfo[indexL];
415 indexL++;
416 indexM++;
417 }
418
419 // Merge Remaining Rights
420 while (indexR < sub2) {
421 info[(indexM + start) % cap] = rinfo[indexR];
422 indexR++;
423 indexM++;
424 }
425
426 // Clean Up and Return
427 delete[] linfo;
428 delete[] rinfo;
429 return;
430 };

487 elmtype KthSmallest(elmtype *arr, int l, int r, int k) {
488 elmtype pivot = arr[(l + r) / 2];
489 int pos = partition(arr, l, r, pivot);
490
491 if (pos - l == k - 1) {
492 return arr[pos];
493 } else if (pos - l > k - 1) {
494 return KthSmallest(arr, l, pos - 1, k);
495 } else {
496 return KthSmallest(arr, pos + 1, r, k - pos + l - 1);
497 }
498 }

512 elmtype KthSmallestWC(elmtype *arr, int l, int r, int k) {
513 int n = r - l + 1, i;
514 elmtype median[(n + 4) / 5];
515 for (i = 0; i < n / 5; i++) {
516 median[i] = getMedian(arr + l + i * 5, 5);
517 }
518 if (n > i * 5) {
519 median[i] = getMedian(arr + l + i * 5, n % 5);
520 i++;
521 }
522 elmtype medOfMed =
523 (i == 1) ? median[i - 1] : KthSmallestWC(median, 0, i - 1, i / 2);
524 int pos = partition(arr, l, r, medOfMed);
525
526 if (pos - l == k - 1) {
527 return arr[pos];
528 } else if (pos - l > k - 1) {
529 return KthSmallestWC(arr, l, pos - 1, k);
530 } else {
531 return KthSmallestWC(arr, pos + 1, r, k - pos + l - 1);
532 }
533 }

540 elmtype getMedian(elmtype *arr, int n) {
541 sort(arr, n);
542 return arr[n / 2];
543 }

557 int partition(elmtype arr[], int l, int r, elmtype pivot) {
558 int i;
559 for (i = l; i < r; i++) {
560 if (arr[i] == pivot) {
561 break;
562 }
563 }
564 swap(&arr[i], &arr[r]);
565 i = l;
566 for (int j = l; j <= r - 1; j++) {
567 if (arr[j] <= pivot) {
568 swap(&arr[i], &arr[j]);
569 i++;
570 }
571 }
572 swap(&arr[i], &arr[r]);
573 return i;
574 }

581 void sort(elmtype *arr, int n) {
582 for (int i = 0; i < n; i++) {
583 elmtype *smallest = &arr[0];
584 for (int j = i + 1; j < n; j++) {
585 if (arr[j] < (*smallest)) {
586 smallest = &arr[j];
587 }
588 }
589 if (&arr[i] != smallest) {
590 swap(&arr[i], smallest);
591 }
592 }

715 void siftDown(int i) {
716 int l = lIndex(i), r = rIndex(i), min = i;
717
718 if (l < info->length() && (*info)[l] < (*info)[i]) { min = l; }
719 if (r < info->length() && (*info)[r] < (*info)[min]) { min = r; }
720 if (min != i) { swap(i, min); siftDown(min); }
