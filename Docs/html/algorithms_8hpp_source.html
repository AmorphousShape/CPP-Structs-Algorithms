171 __uint128_t fib(int *n) {
172
173 // Check if number is too large for __uint128_t
174 if (*n > 185) {
175 throw runtime_error("Fibonacci number too large for unsigned 128-bit int\n");
176 }
177
178 if (*n == 0) {
179 return 0;
180 }
181
182 // Check if number is stored
183 if (fibList[*n] != 0) {
184 return fibList[*n];
185 }
186
187 // If number is not stored, calculate it
188 int a = *n - 1;
189 int b = *n - 2;
190 fibList[*n] = fib(&a) + fib(&b);
191
192 return fibList[*n];
193 }
194
195 public:
196
202 Fib() {
203 // Set first two fibonacci numbers and clear all other data in map
204 fibList[0] = 0;
205 fibList[1] = 1;
206 for (int i = 2; i < 186; i++) {
207 fibList[i] = 0;
208 }
209 }

216 void print(int n, ostream &out = cout) {
217
218 __uint128_t fibNum = fib(&n);
219 if (fibNum == 0) {
220 out << "fib(" << n << ") = 0\n";
221 return;
222 }
223
224 string fibString;
225 while (fibNum > 0) {
226 fibString.insert(fibString.begin(), '0' + (fibNum % 10));
227 fibNum /= 10;
228 }
229
230 out << "fib(" << n << ") = " << fibString << endl;
231 }

294 void build() {
295
296 // Table building loop
297 for (int j = 0; j <= n; j++) {
298 for (int k = 0; k <= max; k++) {
299 if (j == 0) {
300 table[j][k] = 0;
301 } else if (k < weights[j]) {
302 table[j][k] = table[j - 1][k];
303 } else {
304 // max() function is not functioning properly, so I manually find the max
305 int
306 a = table[j - 1][k],
307 b = table[j - 1][k - weights[j]] + profits[j];
308 if (a > b) {
309 table[j][k] = a;
310 } else {
311 table[j][k] = b;
312 }
313 }
314 }
315 }
316
317 // After table is built, choose the objects and set built to true
318 choose();
319 built = true;
320 }
321
325 void choose() {
326
327 // Start at the bottom right corner of the table and work backwards
328 int k = max;
329 for (int i = n; i > 0; i--) {
330 if (table[i-1][k] == table[i][k]) {
331 objects[i] = false;
332 } else {
333 objects[i] = true;
334 k -= weights[i];
335 }
336 }
337 }

463 }

465 return build(j - 1, k - 1) + s1[j - 1];

503 int m = s2.length();
504 L = vector<vector<int>>(n + 1, vector<int>(m + 1, 0));
505
506 for (int j = 1; j <= n; j++) {
507 for (int k = 1; k <= m; k++) {
508 if (s1[j - 1] == s2[k - 1]) {
509 L[j][k] = L[j - 1][k - 1] + 1;

688 void calcCosts() {
689 for (int L = 1; L < n; L++) {
690 for (int i = 0; i < n - L; i++) {
691 int j = i + L;
692 cost->data[i][j] = INT_MAX;
693 for (int k = i; k < j; k++) {
694 int q = cost->data[i][k] + cost->data[k + 1][j] + dim[i] * dim[k + 1] * dim[j + 1];
695 if (q < cost->data[i][j]) {
696 cost->data[i][j] = q;
697 bestK->data[i][j] = k;
698 }
699 }
700 }
701 }

707 Matrix *matrixMult(Matrix *a, Matrix *b) {
708 Matrix *c = new Matrix(a->row, b->col);
709 for (int i = 0; i < a->row; i++) {
710 for (int j = 0; j < b->col; j++) {
711 for (int k = 0; k < a->col; k++) {
712 c->data[i][j] += a->data[i][k] * b->data[k][j];
713 }
714 }
715 }
716 return c;
717 }

727 Matrix *chainProduct(int i, int j) {
728 if (i == j) {
729 return data[i];
730 }
731 int k = bestK->data[i][j];
732 Matrix *a = chainProduct(i, k);
733 Matrix *b = chainProduct(k + 1, j);
734 return matrixMult(a, b);
735 }

808vector<vector<double>> worldSeries(int n, double *aProb) {
809
810 // Create matrix (Use vector as variable length array is not allowed in most compilers)
811 vector<vector<double>> x(n + 1, vector<double>(n + 1, 0.0));
812
813 for (int i = 0; i <= n; i++) {
814 for (int j = 0; j <= n; j++) {
815
816 // Base cases
817 if (i == 0 && j == 0) {
818 x[0][0] = 1;
819 } else if (i == n && j == n) {
820 x[n][n] = 0;
821
822 // Fill in matrix
823 } else if ((i != 0 && j == 0) || (i == n && j != n)) {
824 x[i][j] = x[i - 1][j] * aProb[i + j];
825 } else if ((i == 0 && j != 0) || (i != n && j == n)) {
826 x[i][j] = x[i][j - 1] * (1 - aProb[i + j]);
827 } else {
828 x[i][j] = x[i - 1][j] * aProb[i + j] + x[i][j - 1] * (1 - aProb[i + j]);
829 }
830 }
831 }
832
833 return x;
